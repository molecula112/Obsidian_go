```go
package logger  
  
import (  
    "account/internal/config"  
    "fmt"   
    "os"    
    "strings"  
    
    "github.com/davecgh/go-spew/spew"    
    "github.com/fatih/color"    
    "github.com/rs/zerolog"
)  
  
func New(cfg *config.Config) zerolog.Logger {  
    writer := zerolog.ConsoleWriter{  
       Out:        os.Stdout,  
       TimeFormat: "2026-03-08 13:02:01",  
       FormatLevel: func(i interface{}) string {  
          if i == nil {  
             return ""  
          }  
          level := strings.ToUpper(fmt.Sprintf("%s", i))  
          switch level {  
          case "INFO":  
             return color.New(color.FgGreen).Sprintf("INFO")  
          case "WARN":  
             return color.New(color.FgYellow).Sprintf("WARN")  
          case "ERORR":  
             return color.New(color.FgRed).Sprintf("ERROR")  
          case "DEBUG":  
             return color.New(color.FgBlue).Sprintf("DEBUG")  
          default:  
             return level  
          }  
       },  
    }  
  
    log := zerolog.New(writer).  
       With().  
       Timestamp().  
       Str("source", cfg.ServiceName).  
       Str("env", cfg.AppEnv).  
       Logger()  
  
    spew.Config.Indent = "  "  
    spew.Config.DisablePointerAddresses = true  
    spew.Config.DisableCapacities = true  
    spew.Config.SortKeys = true  
  
    dump := spew.Sdump(cfg)  
  
    log.Info().Msg(dump)  
  
    return log  
  
}
```


```go
func main() {  
    logger := logger.New(cfg)   
}
```
