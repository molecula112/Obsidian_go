```go
package main  
  
import (  
    "account/internal/config"  
    "account/internal/logger"    
    "account/internal/repository"    
    "log"  
    "github.com/gin-gonic/gin"    
)  
   
func main() {  
    router := gin.Default()  
    err = router.SetTrustedProxies([]string{"127.0.0.1"})  
    if err != nil {  
       logger.Error().Msgf("Не удалось установить доверенные прокси: %v", err)  
    }  
  
    router.GET("/ping", PingExample)  
    
    err = router.Run(":8080")  
    if err != nil {  
       return  
    }  
}  
```

Короче главная команда по запуску сервера:
```go
err = router.Run(":8080")  
```
Если посмотришь, то под капотом все тот же `ListenAndServe()`
Также, аналог `ServeMux` - `router := gin.Default() `

Засетапить ендпоинты можно:
```
router.GET("/ping", PingExample) 
```
